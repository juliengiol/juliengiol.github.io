Par la règle de dérivation en chaîne, on obtient donc

\begin{align*} \frac{\mathrm{d}D}{\mathrm{d}t}&=\frac{\partial D}{\partial L}\cdot\frac{\mathrm{d}L}{\mathrm{d}t}+\frac{\partial D}{\partial l}\cdot\frac{\mathrm{d}l}{\mathrm{d}t}+\frac{\partial D}{\partial h}\cdot\frac{\mathrm{d}h}{\mathrm{d}t}\\ &=\frac{1}{2\sqrt{L^2+l^2+h^2}}\cdot 2L\cdot\frac{\mathrm{d}L}{\mathrm{d}t}+\frac{1}{2\sqrt{L^2+l^2+h^2}}\cdot 2l\cdot\frac{\mathrm{d}l}{\mathrm{d}t}\\ &\qquad\qquad+\frac{1}{2\sqrt{L^2+l^2+h^2}}\cdot 2h\cdot\frac{\mathrm{d}h}{\mathrm{d}t}\\ &=\frac{1}{\sqrt{L^2+l^2+h^2}}\cdot\left(L\frac{\mathrm{d}L}{\mathrm{d}t}+l\frac{\mathrm{d}l}{\mathrm{d}t}+h\frac{\mathrm{d}h}{\mathrm{d}t}\right)\\ &=\frac{1}{\sqrt{1^2+2^2+2^2}}\cdot\left(1\cdot 2+2\cdot 2+2\cdot(-3)\right)\\ &=\frac{1}{\sqrt{9}}\cdot 0\\ &=0\,\mathrm{m}/\mathrm{s}. \end{align*}
