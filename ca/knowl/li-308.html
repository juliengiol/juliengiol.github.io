\begin{equation*} x=0 \end{equation*}

\begin{equation*} A=(0,1,f(0,1))\quad\text{et}\quad B=(0,2,f(0,2)). \end{equation*}

\begin{equation*} f_y(0,1)>0. \end{equation*}

\begin{equation*} f_y(0,2)<0. \end{equation*}
