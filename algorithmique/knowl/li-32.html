Exécutez le code suivant à la main pour déterminer ce qu'il affiche à la sortie.

# pour pouvoir utiliser la fonction racine carrée
# sqrt du module math
from math import sqrt

a = 1.0
b = -6.0
c = 9.0
delta = b ** 2 - 4 * a * c
if delta > 0:
    x1 = (- b - sqrt(delta)) / (2 * a)
    x2 = (- b + sqrt(delta)) / (2 * a)
    print(x1, x2)
elif delta == 0:
    x0  = - b / (2 * a)
    print(x0)
else:
    print('pas de racines')
