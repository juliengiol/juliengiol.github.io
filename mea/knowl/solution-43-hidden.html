# entrées :
def f(x):
    return x ** 2 - 2
a = 0.0
b = 2.0
longueurMax = 0.01
# instructions :
while b - a > longueurMax:
    print(a, b) # ligne ajoutée
    m = (a + b) / 2
    if f(m) == 0:
        a = m
        b = m
        break
    elif f(m) * f(a) < 0:
        b = m
    else:
        a = m
# sortie :
if a == b:
    print(a)
else:
    print(a, b)
