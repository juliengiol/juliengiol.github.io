  1. def combienDeDeux1(n):
    compteur = 0
    while n % 2 == 0:
        compteur = compteur + 1
        n = n / 2
    return compteur

print(combienDeDeux1(45))
print(combienDeDeux1(24))

  2. def combienDeDeux2(n):
    if n % 2 != 0:
        return 0
    else:
        return 1 + combienDeDeux2(n / 2)

print(combienDeDeux2(45))
print(combienDeDeux2(24))
