Voici une façon de faire :

# entrées :
import numpy as np
def f(x):
    return 4 * np.sqrt(1 - x ** 2)
a = 0.0
b = 1.0
n = 1000 # ligne modifiée
# instructions :
approximation = 0
deltaX = (b - a) / n
x = b # ligne modifiée
while x > a: # ligne modifiée
    approximation = approximation + f(x)
    x = x - deltaX # ligne modifiée
approximation = deltaX * approximation
# sortie :
print(approximation)
