def natureTriangle(a, b, c):
    if (a > b + c) or (b > a + c) or (c > a + b):
        return 'impossible'
    elif (a == b) and (a == c):
        return 'équilatéral'
    elif (a == b) or (a == c) or (b == c):
        return 'isocèle'
    else:
        return 'scalène'

print(natureTriangle(1, 2, 4))
print(natureTriangle(2, 2, 2))
print(natureTriangle(2, 2, 3))
print(natureTriangle(1, 2, 2.5))
