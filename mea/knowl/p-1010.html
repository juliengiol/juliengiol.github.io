def rectanglesAuMilieu(f, a, b, n):
    approximation = 0
    deltaX = (b - a) / n
    x = a + deltaX / 2
    while x < b:
        approximation = approximation + f(x)
        x = x + deltaX
    approximation = deltaX * approximation
    return approximation
