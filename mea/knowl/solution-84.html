# fonction
def rectanglesAuMilieu(f, a, b, n):
    approximation = 0
    deltaX = (b - a) / n
    x = a + deltaX / 2
    while x < b:
        approximation = approximation + f(x)
        x = x + deltaX
    approximation = deltaX * approximation
    return approximation

# test
def f(x):
    return 1 / (1 + x)
print(rectanglesAuMilieu(f, 0.0, 1.0, 1000))
