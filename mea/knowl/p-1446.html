\begin{align*} \alpha\cdot d=\beta\cdot d\cdot q+r&\quad\Rightarrow\quad r=\alpha\cdot d-\beta\cdot d\cdot q\\ &\quad\Rightarrow\quad r=(\alpha-\beta\cdot q)\cdot d\\ &\quad\Rightarrow\quad r=\delta\cdot d\quad\text{avec }\delta=\alpha-\beta\cdot q\in\mathbb{Z}\\ &\quad\Rightarrow\quad d\vert r. \end{align*}
