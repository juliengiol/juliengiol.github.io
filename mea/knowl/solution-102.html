Voici une façon de faire :

# entrées :
import numpy as np
def f(x):
    return 4 * np.sqrt(1 - x ** 2)
a = 0.0
b = 1.0
n = 32
# instructions :
approximation = 0
deltaX = (b - a) / n
xi1 = a
while xi1 < b:
    xi2 = xi1 + deltaX
    approximation = approximation + f(xi1) + f(xi2)
    xi1 = xi1 + deltaX
approximation = approximation * deltaX / 2
# sortie :
print(approximation)
