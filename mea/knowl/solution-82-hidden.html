import numpy as np
def f(x):
    return 4 * np.sqrt(1 - x ** 2)
a = 0.0
b = 1.0
n = 1000

# instructions
approximation = 0
deltaX = (b - a) / n
x = b
while x > a:
    approximation = approximation + f(x)
    x = x - deltaX
approximation = deltaX * approximation

# sortie
print(approximation)
