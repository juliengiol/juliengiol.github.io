Voici une façon de faire :

# entrées :
import numpy as np
def f(x):
    return 4 * np.sqrt(1 - x ** 2)
a = 0.0
b = 1.0
n = 32
# instructions :
approximation = 0
deltaX = (b - a) / n
for k in range(1, n+1):
    x = a + (k - 1) * deltaX
    approximation = approximation + f(x)
approximation = deltaX * approximation
# sortie :
print(approximation)
