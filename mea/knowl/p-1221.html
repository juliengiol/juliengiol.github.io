\begin{equation*} \begin{array}{cccc} &|\text{Erreur}|&\leq&10^{-6}\\ \Leftrightarrow&\frac{M(b-a)^2}{2n}&\leq&10^{-6}\\ \Leftrightarrow&\frac{1\cdot(1-0)^2}{2n}&\leq&10^{-6}\\ \Leftrightarrow&\frac{1}{2n}&\leq&10^{-6}\\ \Leftrightarrow&\frac{10^6}{2}&\leq&n\\ \Leftrightarrow&500\,000&\leq&n \end{array} \end{equation*}
