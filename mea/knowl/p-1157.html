def trapezes(f, a, b, n):
    approximation = 0
    deltaX = (b - a) / n
    for k in range(1, n+1):
        xi1 = a + (k - 1) * deltaX
        xi2 = xi1 + deltaX
        approximation = approximation + f(xi1) + f(xi2)
    approximation = approximation * deltaX / 2
    return approximation
