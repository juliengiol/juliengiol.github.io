# entrées :
def f(x):
    return x ** 2 - 2
a = 0.0
b = 2.0
longueurMax = 0.01
# instructions :
ya = f(a) # ligne ajoutée
while b - a > longueurMax:
    m = (a + b) / 2
    ym = f(m) # ligne ajoutée
    if ym == 0: # ligne modifiée
        a = m
        b = m
        break
    elif ym * ya < 0: # ligne modifiée
        b = m
    else:
        a = m
        ya = ym # ligne ajoutée
# sortie :
if a == b:
    print(a)
else:
    print(a, b)
