\begin{align*} \text{Aire}&\quad=\quad\int_1^{e^3}\ln x\,\mathrm{d}x\\ &\quad\stackrel{\text{IPP}}{=}\quad x\ln x\Bigg|_1^{e^3}-\int_1^{e^3}\bcancel{x}\cdot\frac{1}{\bcancel{x}}\,\mathrm{d}x\\ &\quad=\quad x\ln x-x\Bigg|_1^{e^3}\\ &\quad=\quad (e^3\underbrace{\ln\left(e^3\right)}_{=\,3}-e^3)-(1\underbrace{\ln(1)}_{=\,0}-1)\\ &\quad=\quad 3e^3-e^3+1\\ &\quad=\quad 2e^3+1 \end{align*}
